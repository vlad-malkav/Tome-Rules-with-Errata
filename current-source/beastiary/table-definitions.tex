
%%%New monster table environments for consistency%%%
\newenvironment{monstersingle}{
\rowcolors{0}{white}{tablegrey}
\begin{longtabu} to \linewidth{X X}
\toprule
}{
\bottomrule
\end{longtabu}
}

\newenvironment{monsterdouble}{
\rowcolors{0}{white}{tablegrey}
\begin{longtabu} to \linewidth{X X X X}
\toprule
}{
\bottomrule
\end{longtabu}
}

%%%Generic row formats for consistency%%%
\newcommand{\singlemonsterline}[4]{{\bf #1} #3 & {\bf #2} #4 \\}

\newcommand{\singlemonsterspanline}[2]{\multicolumn{2}{p{0.975\linewidth\hangindent=0.3cm}}{\raggedright{{\bf #1} #2}} \\}

\newcommand{\singlemonsterabilityline}[2]{\multicolumn{2}{p{0.975\linewidth\hangindent=0.3cm}}{{\bf #1:} #2} \\}

\newcommand{\doublemonsterline}[6]{{\bf #1} #3 & {\bf #2} #4 & {\bf #1} #5 & {\bf #2} #6 \\}

\newcommand{\doublemonsterspanline}[3]{\multicolumn{2}{p{0.475\linewidth\hangindent=0.3cm}}{\raggedright{{\bf #1} #2}} & \multicolumn{2}{p{0.475\linewidth\hangindent=0.3cm}}{\raggedright{{\bf #1} #3}} \\}

\newcommand{\doublemonsterabilityline}[2]{\multicolumn{4}{p{0.975\linewidth\hangindent=0.3cm}}{{\bf #1:} #2} \\}
