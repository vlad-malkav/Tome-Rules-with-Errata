%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\raceentry{Half-Elf}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\quot{``I don't fit in anywhere, please, listen to me cry.''}

\begin{itemize}
\item \racesize{Medium}
\item \racetype{Humanoid (Human, Elf)}
\item \racemovement{30ft}
\item \racevision{Low-light vision}
\item Immunity to Sleep effects, and a +2 racial bonus on saving throws against Enchantment spells or effects.
\item +1 racial bonus on \linkskill{Listen}, \linkskill{Search}, and \linkskill{Spot} checks.
\item +2 racial bonus on \linkskill{Diplomacy} and \linkskill{Gather Information} checks.
\item \racefavoredclass{Any}
\item \racelang{Common}
\item \raceextralang{Any}
\end{itemize}
