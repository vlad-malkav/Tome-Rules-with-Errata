\subsection{Mechanics}
%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{itemize*}
\item \racesize{Medium}
\item \racetype{Outsider (Native, Human)}
\item \racemovement{30ft}
\item \racevision{Darkvision 60ft}
\item +2 Charisma, +2 Wisdom
\item Aasimar with a Charisma of at least 10 may cast \linkspell{Light} as a spell-like ability at will. They may only have one such Light active at a time.
\item +2 racial bonus to \linkskill{Spot}, and \linkskill{Listen} checks.
\item \racefavoredclass{Any class with Charisma-based spellcasting ability}
\item \racelang{Common}
\item \raceextralang{Abyssal, Aquan, Auran, Celestial, Formian, Giant Frog, Ignan, Infernal, Sylvan, Terran}
\end{itemize*}
