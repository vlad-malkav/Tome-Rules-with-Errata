%%%%%%%%%%%%%%%%%%%%%%%%%
\subsection{Mechanics}
%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{itemize*}
\item \racesize{Medium}
\item \racetype{Humanoid (Elf)}
\item \racemovement{30ft}
\item \racevision{Darkvision 120ft}
\item +2 Dexterity, -2 Constitution
\item \textbf{Extreme Daylight Sensitivity (Ex):} Drow are \linkcondition{Dazzled} in bright sunlight or within the radius of a \linkspell{Daylight} spell, but with twice the normal effect (-2 instead of -1).
\item Drow with a Charisma of at least 10 may cast \linkspell{Deeper Darkness} (duration 4 hours), and \linkspell{Faerie Fire} as spell-like abilities once per day each.
\item +2 bonus to saving throws against spells and spell-like abilities.
\item +2 bonus to \linkskill{Spot}, and \linkskill{Listen} checks.
\item Drow never sleep and are immune to Sleep effects. Drow must still perform their 4 hour daily trance to stay coherent and rested.
\item Drow live an exceedingly interesting life and every Drow has proficiency with the rapier and an exotic ranged weapon of their choice.
\item \racefavoredclass{Any spellcasting class}
\item \racelang{Elvish}
\item \raceextralang{Abyssal, Beholder, Common, Draconic, Drow Sign Language, Dwarvish, Gnome, Kuo-Toa, Terran, Undercommon}
\end{itemize*}
